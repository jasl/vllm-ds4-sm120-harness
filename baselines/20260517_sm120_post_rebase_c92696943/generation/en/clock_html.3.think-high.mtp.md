                const dist = numberDist;
                for (let i = 1; i <= 12; i++) {
                    const el = document.createElement('div');
                    el.className = 'number';
                    el.textContent = i;
                    const angle = i * 30;
                    el.style.transform = 'rotate(' + angle + 'deg) translateY(-' + dist + 'px) rotate(-' + angle + 'deg)';
                    numbersContainer.appendChild(el);
                }
            }

            // ---- Beijing Time (UTC+8) ----
            function getBeijingTime() {
                const now = new Date();
                const localOffset = now.getTimezoneOffset(); // minutes, e.g. -480 for UTC+8
                const beijingOffset = -480; // UTC+8 = -480 minutes from UTC
                const diffMs = (beijingOffset - localOffset) * 60000;
                return new Date(now.getTime() + diffMs);
            }

            // ---- Update clock ----
            function updateClock() {
                const bt = getBeijingTime();

                const h = bt.getHours();
                const m = bt.getMinutes();
                const s = bt.getSeconds();
                const ms = bt.getMilliseconds();

                // Smooth angles
                const secondAngle = (s + ms / 1000) * 6;
                const minuteAngle = (m + s / 60) * 6;
                const hourAngle = ((h % 12) + m / 60) * 30;

                // Apply rotations
                hourHand.style.transform = 'rotate(' + hourAngle + 'deg)';
                minuteHand.style.transform = 'rotate(' + minuteAngle + 'deg)';
                secondHand.style.transform = 'rotate(' + secondAngle + 'deg)';

                // Time text (24-hour format)
                const timeStr = String(h).padStart(2, '0') + ':' +
                    String(m).padStart(2, '0') + ':' +
                    String(s).padStart(2, '0');
                timeText.textContent = timeStr;

                // Date and weekday
                const weekdays = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
                const dayName = weekdays[bt.getDay()];
                const year = bt.getFullYear();
                const month = String(bt.getMonth() + 1).padStart(2, '0');
                const day = String(bt.getDate()).padStart(2, '0');
                dateText.textContent = year + '-' + month + '-' + day + '  ' + dayName;
            }

            // ---- Animation loop ----
            function animate() {
                updateClock();
                animationId = requestAnimationFrame(animate);
            }

            // ---- Resize handler ----
            let resizeTimer = null;

            function handleResize() {
                if (resizeTimer) {
                    cancelAnimationFrame(resizeTimer);
                    resizeTimer = null;
                }
                resizeTimer = requestAnimationFrame(function() {
                    rebuildLayout();
                    resizeTimer = null;
                });
            }

            // ---- Init ----
            function init() {
                // Initial dimension calc
                recalcDimensions();
                buildScales();
                buildNumbers();

                // Start animation
                if (animationId) {
                    cancelAnimationFrame(animationId);
                }
                animate();

                // Resize listener
                window.addEventListener('resize', handleResize);

                // Also observe the container for layout changes (edge case)
                if (window.ResizeObserver) {
                    const ro = new ResizeObserver(function() {
                        handleResize();
                    });
                    ro.observe(clockFace);
                }
            }

            // ---- Start when DOM ready ----
            if (document.readyState === 'loading') {
                document.addEventListener('DOMContentLoaded', init);
            } else {
                init();
            }

            // ---- Cleanup on page unload (optional) ----
            window.addEventListener('beforeunload', function() {
                if (animationId) {
                    cancelAnimationFrame(animationId);
                    animationId = null;
                }
            });

        })();
    </script>

</body>

### Clock Design

The clock face is built with CSS and JavaScript for a clean, high-end look. Here’s how it works:

- **Time & Date Display**: The upper half shows the current time in 24-hour format (HH:MM:SS), while the lower half displays the date and weekday. Both are absolutely centered for a clean layout.
- **Hands & Movement**: The hour and minute hands are black and slender, while the second hand is red with a subtle tail. The second hand rotates smoothly (using milliseconds) for a fluid, analog feel.
- **Scale & Numbers**: 60 scale marks are generated dynamically. Hour marks (every 5th) are thicker and darker, while minute marks are thinner and lighter. Numbers 1–12 are positioned around the dial using rotation transforms, keeping them upright.
- **Visual Style**: The clock is a white circle with a dark rounded border and a 3D shadow effect. The sans-serif typography and minimalist color scheme create a modern, high-end wall clock aesthetic.

The clock automatically updates every frame using `requestAnimationFrame`, ensuring smooth second-hand motion and accurate Beijing time.
