                    const x = NUM_RADIUS * Math.sin(rad);
                    const y = -NUM_RADIUS * Math.cos(rad);

                    const el = document.createElement('div');
                    el.className = 'number';
                    el.textContent = i;
                    el.style.transform = 'translate(-50%, -50%) translate(' + x + 'px, ' + y + 'px)';
                    // reset any inherited transform-origin
                    el.style.left = '50%';
                    el.style.top = '50%';
                    frag.appendChild(el);
                }
                face.appendChild(frag);
            }

            // ── build hands ──
            function buildHands() {
                const hour = document.createElement('div');
                hour.className = 'hand hand-hour';
                hour.id = 'handHour';

                const minute = document.createElement('div');
                minute.className = 'hand hand-minute';
                minute.id = 'handMinute';

                const second = document.createElement('div');
                second.className = 'hand hand-second';
                second.id = 'handSecond';

                face.appendChild(hour);
                face.appendChild(minute);
                face.appendChild(second);

                // centre pin (on top of hands)
                const pin = document.createElement('div');
                pin.className = 'centre-pin';
                face.appendChild(pin);
            }

            // ── build info texts ──
            function buildTexts() {
                const top = document.createElement('div');
                top.className = 'info-text info-top';
                top.id = 'infoTop';

                const bottom = document.createElement('div');
                bottom.className = 'info-text info-bottom';
                bottom.id = 'infoBottom';

                face.appendChild(top);
                face.appendChild(bottom);
            }

            // ── initialise DOM ──
            buildScales();
            buildNumbers();
            buildHands();
            buildTexts();

            // grab hand elements
            const hourHand = document.getElementById('handHour');
            const minuteHand = document.getElementById('handMinute');
            const secondHand = document.getElementById('handSecond');
            const infoTop = document.getElementById('infoTop');
            const infoBottom = document.getElementById('infoBottom');

            // ── time helpers ──
            function getBeijingTime() {
                // Use toLocaleString with Asia/Shanghai to get a reliable string
                const str = new Date().toLocaleString('en-US', {
                    timeZone: 'Asia/Shanghai',
                    hour12: false,
                    year: 'numeric',
                    month: '2-digit',
                    day: '2-digit',
                    hour: '2-digit',
                    minute: '2-digit',
                    second: '2-digit',
                });
                // str → "MM/DD/YYYY, HH:MM:SS"
                const parts = str.split(', ');
                const datePart = parts[0]; // "MM/DD/YYYY"
                const timePart = parts[1]; // "HH:MM:SS"

                const [month, day, year] = datePart.split('/').map(Number);
                const [h, m, s] = timePart.split(':').map(Number);

                // Also grab milliseconds from local Date, then adjust to Beijing
                const now = new Date();
                const localMs = now.getTime();
                const offsetLocal = now.getTimezoneOffset() * 60000;
                const beijingOffset = 8 * 3600000;
                const beijingMs = localMs + offsetLocal + beijingOffset;
                const ms = beijingMs % 1000;

                return { year, month, day, hours: h, minutes: m, seconds: s, ms };
            }

            function getBeijingDate() {
                // Build a Date object for Beijing time zone to get day-of-week
                const now = new Date();
                const offsetLocal = now.getTimezoneOffset() * 60000;
                const beijingOffset = 8 * 3600000;
                const beijingMs = now.getTime() + offsetLocal + beijingOffset;
                const beijingDate = new Date(beijingMs);

                const days = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
                const months = [
                    'January', 'February', 'March', 'April', 'May', 'June',
                    'July', 'August', 'September', 'October', 'November', 'December'
                ];

                const y = beijingDate.getFullYear();
                const m = months[beijingDate.getMonth()];
                const d = beijingDate.getDate();
                const w = days[beijingDate.getDay()];

                return { dateStr: d + ' ' + m + ' ' + y, weekStr: w };
            }

            // ── update clock ──
            function updateClock() {
                const t = getBeijingTime();
                const { hours, minutes, seconds, ms } = t;

                // smooth second angle (continuous)
                const smoothSec = seconds + ms / 1000;
                const secAngle = smoothSec * 6;

                // minute angle (continuous via seconds)
                const minAngle = minutes * 6 + (seconds / 60) * 6 + (ms / 60000) * 6;

                // hour angle (continuous via minutes & seconds)
                const hourAngle = (hours % 12) * 30 + (minutes / 60) * 30 + (seconds / 3600) * 30;

                // apply rotations
                hourHand.style.transform = 'rotate(' + hourAngle + 'deg)';
                minuteHand.style.transform = 'rotate(' + minAngle + 'deg)';
                secondHand.style.transform = 'rotate(' + secAngle + 'deg)';

                // ── text info ──
                // top: 24-hour time with leading zeros
                const hh = String(hours).padStart(2, '0');
                const mm = String(minutes).padStart(2, '0');
                const ss = String(seconds).padStart(2, '0');
                infoTop.textContent = hh + ':' + mm + ':' + ss;

                // bottom: date + day of week
                const dateInfo = getBeijingDate();
                infoBottom.textContent = dateInfo.weekStr + ' · ' + dateInfo.dateStr;

                // smooth loop
                requestAnimationFrame(updateClock);
            }

            // ── start ──
            updateClock();

        })();
    </script>

</body>
</html>

### Clock Interface

The clock combines analog readability and digital clarity for quick time checks.

- **Analog Dial & Hands**: The circular dial features 60 scale marks—thicker at each hour, thinner at minute intervals. The hour and minute hands are slender and dark, while the second hand is red with a small counterweight for a refined look.
- **Smooth Second Hand**: Unlike a ticking motion, the second hand moves continuously using milliseconds, creating a fluid, premium feel.
- **Digital Time & Date**: The upper half displays the current time in 24-hour format (HH:MM:SS). The lower half shows the day of the week and full date (e.g., “Monday · 14 April 2025”), both centered and using a clean sans-serif font.
- **Beijing Time Zone**: All time values are converted to China Standard Time (UTC+8) using the `Asia/Shanghai` time zone, ensuring accuracy regardless of your local system settings.

---

**Optimization Tip:** You can adjust the clock’s visual size by modifying the `.clock` width/height (default 420px) or the `RADIUS` constant (line ~80) to scale all elements proportionally.
