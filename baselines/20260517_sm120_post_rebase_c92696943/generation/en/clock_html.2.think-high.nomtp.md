````markdown
                // 创建 60 个刻度
                for (let i = 0; i < TICK_COUNT; i++) {
                    const angle = i * 6; // 每个刻度 6°
                    const isHour = (i % HOUR_TICK_INTERVAL === 0);

                    const tick = document.createElement('div');
                    tick.className = 'tick' + (isHour ? ' tick-hour' : '');
                    // 用 transform 旋转
                    tick.style.transform = 'rotate(' + angle + 'deg)';
                    face.appendChild(tick);
                }

                // 创建 12 个数字 (1-12)
                const numberRadius = radius * 0.84;
                for (let i = 1; i <= 12; i++) {
                    // 角度：12点方向为 0°，顺时针递增
                    // 数字 i 对应角度 = i * 30°
                    const angleDeg = i * 30;
                    const rad = (angleDeg - 90) * Math.PI / 180;
                    const x = Math.cos(rad) * numberRadius;
                    const y = Math.sin(rad) * numberRadius;

                    const num = document.createElement('div');
                    num.className = 'number';
                    num.textContent = i;
                    num.style.transform = 'translate(-50%, -50%) translate(' + x + 'px, ' + y + 'px)';
                    face.appendChild(num);
                }
            }

            // 初始构建 & 窗口变化时重建
            buildDial();
            window.addEventListener('resize', buildDial);

            // ----- 获取北京时间的辅助函数 -----
            function getBeijingTime() {
                // 使用 toLocaleString 获取北京时间字符串，再解析
                const str = new Date().toLocaleString('en-US', {
                    timeZone: 'Asia/Shanghai',
                    hour12: false,
                    year: 'numeric',
                    month: '2-digit',
                    day: '2-digit',
                    hour: '2-digit',
                    minute: '2-digit',
                    second: '2-digit'
                });
                // 输出格式: "MM/DD/YYYY, HH:MM:SS"
                const parts = str.split(', ');
                const datePart = parts[0]; // "MM/DD/YYYY"
                const timePart = parts[1]; // "HH:MM:SS"

                const [month, day, year] = datePart.split('/').map(Number);
                const [hour, minute, second] = timePart.split(':').map(Number);

                // 同时获取毫秒（toLocaleString 不包含毫秒，需要从原始 Date 取）
                const now = new Date();
                const ms = now.getMilliseconds();

                return {
                    year,
                    month,
                    day,
                    hour,
                    minute,
                    second,
                    ms,
                    // 星期几 (0=周日, 1=周一 ...)
                    weekday: new Date(year, month - 1, day).getDay()
                };
            }

            // ----- 星期名称 (中文) -----
            const WEEK_NAMES = ['日', '一', '二', '三', '四', '五', '六'];

            // ----- 更新时钟 -----
            function updateClock() {
                const t = getBeijingTime();

                // --- 1. 指针角度 (平滑) ---
                // 秒针: 包含毫秒实现平滑
                const smoothSec = t.second + t.ms / 1000;
                const secAngle = smoothSec * 6; // 每秒 6°

                // 分针: 包含秒
                const smoothMin = t.minute + smoothSec / 60;
                const minAngle = smoothMin * 6; // 每分 6°

                // 时针: 包含分
                const smoothHour = (t.hour % 12) + smoothMin / 60;
                const hourAngle = smoothHour * 30; // 每小时 30°

                // 应用旋转 (禁用过渡以实现即时响应)
                hourHand.style.transition = 'none';
                minuteHand.style.transition = 'none';
                secondHand.style.transition = 'none';

                hourHand.style.transform = 'rotate(' + hourAngle + 'deg)';
                minuteHand.style.transform = 'rotate(' + minAngle + 'deg)';
                secondHand.style.transform = 'rotate(' + secAngle + 'deg)';

                // --- 2. 上半部分：时间文本 (24小时制 HH:MM:SS) ---
                const hh = String(t.hour).padStart(2, '0');
                const mm = String(t.minute).padStart(2, '0');
                const ss = String(t.second).padStart(2, '0');
                timeText.textContent = hh + ':' + mm + ':' + ss;

                // --- 3. 下半部分：日期 & 星期 ---
                const y = t.year;
                const m = String(t.month).padStart(2, '0');
                const d = String(t.day).padStart(2, '0');
                const w = WEEK_NAMES[t.weekday];
                dateText.textContent = y + '年' + m + '月' + d + '日  星期' + w;
            }

            // ----- 启动循环 (requestAnimationFrame) -----
            let lastTime = 0;

            function loop(timestamp) {
                // 每帧更新 (约 16ms 间隔，保证秒针平滑)
                if (timestamp - lastTime >= 16) {
                    updateClock();
                    lastTime = timestamp;
                }
                requestAnimationFrame(loop);
            }

            // 首次立即更新，然后启动循环
            updateClock();
            requestAnimationFrame(loop);

            // ----- 边缘情况: 页面可见性变化时强制刷新 -----
            document.addEventListener('visibilitychange', function() {
                if (!document.hidden) {
                    updateClock();
                }
            });

            // 也支持 setInterval 作为 fallback (但 requestAnimationFrame 已足够)
            // 额外每 1 秒确保同步 (防止标签页休眠导致的偏差)
            setInterval(function() {
                // 仅做修正，不依赖它驱动动画
                const now = Date.now();
                // 如果距离上次更新超过 2 秒，强制更新
                if (now - lastTime > 2000) {
                    updateClock();
                    lastTime = now;
                }
            }, 1000);

        })();
    </script>

</body>
</html>
```
### 极简时钟说明

整个页面会实时同步北京时间，并在一个表盘上同时显示指针指示、数字钟和日历信息。

- **指针系统**：时针和分针为黑色细长造型，秒针采用红色高亮设计并带有尾部配重。秒针实现了平滑扫秒效果，而非传统的跳秒，视觉上更接近高端石英钟。
- **文本信息**：钟面上半部分使用大号无衬线字体显示当前时间（24 小时制，格式为 `HH:MM:SS`），下半部分显示当前日期和中文星期（如 `2025年06月18日 星期三`），信息一目了然。
- **表盘细节**：白色圆形表盘带有深色圆角边框和 3D 阴影效果。60 个刻度中，整点刻度更粗更黑，非整点刻度更细更浅，指针从圆心伸出，整体风格极简且精致。
````
