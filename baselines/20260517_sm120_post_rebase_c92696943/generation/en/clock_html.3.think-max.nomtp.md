                // Rotate to position, translate outward, then rotate back to keep upright
                num.style.transform = 'rotate(' + angle + 'deg) translate(0, -' + radius + 'px) rotate(-' + angle + 'deg)';
                container.appendChild(num);
            }
        }

        // Update the clock display with Beijing time (CST, UTC+8)
        function updateClock() {
            var now = new Date();
            var ms = now.getMilliseconds();

            // Parse the Beijing time string exactly as prompted
            var localeStr = now.toLocaleString('en-US', { timeZone: 'Asia/Shanghai' });
            var parts = localeStr.split(', ');
            var dateStr = parts[0];
            var timeStr = parts[1];

            var dateParts = dateStr.split('/').map(Number);
            var month = dateParts[0];
            var day = dateParts[1];
            var year = dateParts[2]; // retained for completeness

            var timeParts = timeStr.split(' ');
            var time = timeParts[0];
            var modifier = timeParts[1];

            var timeNumbers = time.split(':').map(Number);
            var hours = timeNumbers[0];
            var minutes = timeNumbers[1];
            var seconds = timeNumbers[2];

            // Convert 12-hour format to 24-hour format
            if (modifier === 'PM' && hours !== 12) {
                hours += 12;
            }
            if (modifier === 'AM' && hours === 12) {
                hours = 0;
            }

            // Calculate rotation angles with smooth sub-second movement
            var secondsAngle = (seconds + ms / 1000) * 6;
            var minutesAngle = (minutes + seconds / 60 + ms / 60000) * 6;
            var hoursAngle = ((hours % 12) + minutes / 60 + seconds / 3600 + ms / 3600000) * 30;

            // Update the three hands
            document.getElementById('hourHand').style.transform = 'rotate(' + hoursAngle + 'deg)';
            document.getElementById('minuteHand').style.transform = 'rotate(' + minutesAngle + 'deg)';
            document.getElementById('secondHand').style.transform = 'rotate(' + secondsAngle + 'deg)';

            // Update the digital time text (24-hour format)
            var pad = function(n) { return String(n).padStart(2, '0'); };
            document.getElementById('timeDisplay').textContent = pad(hours) + ':' + pad(minutes) + ':' + pad(seconds);

            // Update the date and day of the week
            var dateOptions = {
                timeZone: 'Asia/Shanghai',
                weekday: 'long',
                month: 'long',
                day: 'numeric'
            };
            var dateStrFormatted = new Intl.DateTimeFormat('en-US', dateOptions).format(now);
            document.getElementById('dateDisplay').textContent = dateStrFormatted;
        }

        // Initialize the clock face elements
        createTicks();
        createNumbers();
        updateClock();

        // Start the smooth continuous loop using requestAnimationFrame
        function tick() {
            updateClock();
            requestAnimationFrame(tick);
        }
        requestAnimationFrame(tick);

    })();
